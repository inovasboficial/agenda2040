function doPost(e) {
  try {
    // Parse do JSON recebido
    const dados = JSON.parse(e.postData.contents);
    
    // Acessa a planilha ativa
    const planilha = SpreadsheetApp.getActiveSpreadsheet();
    const aba = planilha.getActiveSheet();
    
    // Adiciona nova linha com os dados
    aba.appendRow([
      dados.dataHora || new Date().toLocaleString('pt-BR', {timeZone: 'America/Sao_Paulo'}),
      dados.tipo || '',
      dados.conteudo || '',
      dados.audioBase64 || '',
      dados.eixo || '',
      dados.eixoSugerido || '',
      dados.bairro || '',
      dados.latitude || '',
      dados.longitude || '',
      dados.nome || '',
      dados.instituicao || '',
      dados.userAgent || ''
    ]);
    
    // Retorna sucesso com CORS
    return ContentService
      .createTextOutput(JSON.stringify({ 
        status: 'sucesso', 
        mensagem: 'Dados salvos com sucesso!' 
      }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (erro) {
    // Retorna erro
    return ContentService
      .createTextOutput(JSON.stringify({ 
        status: 'erro', 
        mensagem: erro.toString() 
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// OPCIONAL: Converter Base64 em arquivo no Google Drive
function converterAudioBase64ParaDrive() {
  const planilha = SpreadsheetApp.getActiveSpreadsheet();
  const aba = planilha.getActiveSheet();
  const dados = aba.getDataRange().getValues();
  
  // Pula cabeçalho
  for (let i = 1; i < dados.length; i++) {
    const audioBase64 = dados[i][3]; // Coluna D
    if (audioBase64 && audioBase64.startsWith('data:audio')) {
      try {
        // Remove prefixo data:audio/webm;base64,
        const base64 = audioBase64.split(',')[1];
        const blob = Utilities.newBlob(Utilities.base64Decode(base64), 'audio/webm', `audio_contribuicao_${i}.webm`);
        
        // Salva no Drive
        const arquivo = DriveApp.createFile(blob);
        const url = arquivo.getUrl();
        
        // Atualiza célula com link
        aba.getRange(i + 1, 4).setValue(url);
      } catch (erro) {
        Logger.log(`Erro linha ${i + 1}: ${erro}`);
      }
    }
  }
}
